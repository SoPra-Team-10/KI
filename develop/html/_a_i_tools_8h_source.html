49  auto aStarSearch(const SearchNode<T> &start, const SearchNode<T> &destination,
50  const Exp &expand, const F &f) -> std::optional<SearchNode<T>>{
51  std::deque<SearchNode<T>> visited;
52  std::list<SearchNode<T>> fringe = {start};
53  std::optional<SearchNode<T>> ret;
54  while(!fringe.empty()){
55  const auto currentNode = *fringe.begin();
56  fringe.pop_front();
57  if(currentNode == destination){
58  ret.emplace(currentNode);
59  break;
60  }
61 
62  bool nodeVisited = false;
63  for(const auto &node : visited){
64  if(currentNode == node){
65  nodeVisited = true;
66  break;
67  }
68  }
69 
70  if(nodeVisited){
71  continue;
72  }
73 
74  visited.emplace_back(currentNode);
75  for(const auto &newNode : expand(currentNode)){
76  bool inserted = false;
77  for(auto it = fringe.begin(); it != fringe.end(); ++it){
78  if(f(newNode) < f(*it)){
79  fringe.emplace(it, newNode);
80  inserted = true;
81  break;
82  }
83  }
84 
85  if(!inserted){
86  fringe.emplace_back(newNode);
87  fringe.size();
88  }
89  }
90  }
91 
92  return ret;
93  }
